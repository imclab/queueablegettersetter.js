QueuableGetterSetter.js

Like Getter Setter but Queueable

Slides - JsDoc - GitHub

// define an object 
var foo = { bar : 'hello' };
// define a first getter
foo.__defineQGetter__('bar', function(value){
    return value+ ', world.';
});
// define a second getter
foo.__defineQGetter__('bar', function(value){
    return value+ ' I love you!';
});
// display the value of foo.bar
console.log(foo.bar)
// output a love message
// "Hello, world. I love you!"
